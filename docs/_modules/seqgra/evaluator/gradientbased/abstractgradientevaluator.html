    def _backprop(self, inp, ind):
        output = self.learner.model(inp)
        if ind is None:
            ind = output.data.max(1)[1].unsqueeze(0)
        grad_out = output.data.clone()
        grad_out.fill_(0.0)
        grad_out.scatter_(1, ind, 1.0)
        output.backward(grad_out)
        return inp.grad.data
